任务：

做一个这样子的布局样式

如何布局

第一种想法

  1. 如图，整体可以堪称两行一列的表格布局，然后上面的布局里面是是边框布局，上面的中间看作一个两行一列的表格布局，下面的整体也一样

  2. 具体的代码实现如下

    public class Test02 {
    public static void main(String[] args) {
        //new一个Frame
        //设置布局为2行一列的表格
        //设置背景色
        //设置大小
        //设置位置
        //设置可见
        Frame frame = new Frame();
        frame.setLayout(new GridLayout(2,1));
        frame.setBackground(Color.black);
        frame.setSize(600,400);
        frame.setLocation(100,100);
        frame.setVisible(true);

        //上面
        Panel panelUp = new Panel(new BorderLayout());
        //上面左右两个button
        panelUp.add(new Button("btn-02"),BorderLayout.EAST);
        panelUp.add(new Button("btn-01"),BorderLayout.WEST);
        //上面中间的两个并排的,放在一个panel上
        Panel panelUpItem = new Panel(new GridLayout(2, 1));
        panelUpItem.add(new Button("btn-03"));
        panelUpItem.add(new Button("btn-04"));
        //panelUpItem添加到panelUp
        panelUp.add(panelUpItem);

        //下面
        Panel panelDown = new Panel(new BorderLayout());
        //下面左右两个button
        panelDown.add(new Button("btn-09"),BorderLayout.EAST);
        panelDown.add(new Button("btn-10"),BorderLayout.WEST);
        //中间2x2
        Panel panelDownItem = new Panel(new GridLayout(2, 2));
        for (int i = 0; i < 4; i++) {
            panelDownItem.add(new Button("btn-0" + (i + 5)));
        }
        //panelDown添加到panelDownItem
        panelDown.add(panelDownItem);

        //将两个panel添加到frame
        frame.add(panelUp);
        frame.add(panelDown);


   }

}

   public class Test01 {
       public static void main(String[] args) {
           Frame frame = new Frame();
           frame.setBackground(Color.black);
           frame.setBounds(100,100,700,500);
           frame.setLayout(new GridLayout(2,3));
           frame.setVisible(true);

           Panel panel1 = new Panel(new GridLayout(2,1));
           Panel panel2 = new Panel(new GridLayout(2,2));

           panel1.add(new Button("btn-2-1"));
           panel1.add(new Button("btn-2-2"));

           panel2.add(new Button("btn-5-1"));
           panel2.add(new Button("btn-5-2"));

           frame.add(new Button("btn-1"));
           frame.add(new Button("btn-3"));
           frame.add(new Button("btn-4"));
           frame.add(new Button("btn-6"));
           frame.add(panel1,1);
           frame.add(panel2,4);

           frame.addWindowListener(new WindowAdapter() {
               @Override
               public void windowClosing(WindowEvent e) {
                   super.windowClosed(e);
                   System.exit(0);
               }
           });

       }

   }
